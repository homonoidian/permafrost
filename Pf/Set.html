    def ===(object : T) : Bool #

    Same as Set#===.

    reds = Pf::Set["red", "pink", "violet"]
blues = Pf::Set["blue", "azure", "violet"]

both = red = blue = false

case "violet"
when reds & blues
  both = true
when reds
  red = true
when blues
  blue = true
end

both # => true
red  # => false
blue # => false

    def same?(other : Set(T)) #

    Returns true if self and other refer to the same set in memory.
